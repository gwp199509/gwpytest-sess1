1，http协议  请求，响应
请求
 请求报文起始行
   请求方法：用于指定客户端如何访问服务端
     GET  下载
     POST  上传
	 HEAD  类似于GET，仅仅输出响应的头部信息
  URI 统一资源标识符，资源的位置，这个资源在网站站点目录的哪个地方
  协议与版本 HTTP/1.1
 请求头
     字段
	 User-Agent   客户端代理，用什么工具访问网站
	 Host      要访问的目标网站：域名或ip
 空行
 请求报文主体
响应
 响应起始行
   协议与版本
   状态码  数字3位，用于描述服务端是否能处理用户的请求，200ok 404未找到
   状态码描述
 响应头
   Server   显示服务端使用的web服务器及版本
   Content-Type  媒体类型
   Content-length  大小
   Location  跳转之后的新的位置rewrite 301/302
 空行
 响应报文主体
2,状态码
  2xx  正常
  3xx  需要跳转，表示正常
  4xx   异常，客户端问题
  5xx   异常，服务端问题
  200 ok
  301  永久跳转
  302   临时跳转
  304   浏览器缓存
  403   权限拒绝，权限问题或首页文件问题
  404    文件找不到，一般辅助错误日志排查
  500    内部错误
  502    网关错误，一般负载中，请求发送到后面，后面无人处理
  503    服务临时不可用，后端负载异常等情况
  504    网关超时
3，网站访问指标和统计
  IP  访问网站的独立ip数量，公网ip
  PV  页面访问量
  UV  独立访客数量，接近于用户数量
  DAU  每天的活跃用户的数量 日活
  MAU  月活，月活跃用户

	name=nginx stable repo
	baseurl=https://nginx.org/packages/centos/$releasever/$basearch/

  yum -y install nginx
  systemctl enable nginx   systemctl start nginx 
  目录结构
	  /etc/nginx  各种配置文件的目录
	  /etc/nginx/nginx.conf   主配置文件
	  /etc/nginx/conf.d/     子配置文件目录
	  /etc/nginx/conf.d/default.conf   默认的子配置文件
	  /usr/sbin/nginx   nginx命令
	  /usr/share/nginx/html/    默认的站点目录，网站的根目录
	  /var/log/nginx/         访问日志，错误日志目录
	  
	  /etc/logrotate.d/nginx  日志切割
	  /etc/nginx/mime.types   媒体类型
	  /etc/nginx/fastcgi_params  nginx+php
	  /etc/nginx/uwsgi_paramms    nginx+python
	  /usr/lib/systemd/system/nginx.servicxe   systemctl配置文件
	  /var/cache/nginx/   缓存目录
  主配置文件 /etc/nginx/nginx.conf
  user nginx；  worker_processes进程用户
  worker_processes auto;    进程数量
  error_log /var/log/nginx/error.log;  错误日志
  pid /run/nginx.pid;   pid文件
  events {               events区域
    worker_connections 1024;  每个进程最多处理连接数量
}
  http {
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '   指定nginx访问日志格式
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;           日志存放位置

    sendfile            on;
    tcp_nopush          on;
    tcp_nodelay         on;
    keepalive_timeout   65;         超时断开时间65s
    types_hash_max_size 4096;

    include             /etc/nginx/mime.types;     媒体类型
    default_type        application/octet-stream;   默认媒体类型

    # Load modular configuration files from the /etc/nginx/conf.d directory.
    # See http://nginx.org/en/docs/ngx_core_module.html#include
    # for more information.
    include /etc/nginx/conf.d/*.conf;    配置文件位置

    server {
        listen       80;
        listen       [::]:80;
        server_name  _;
        root         /usr/share/nginx/html;

        # Load configuration files for the default server block.
        include /etc/nginx/default.d/*.conf;

        error_page 404 /404.html;
        location = /404.html {
        }

        error_page 500 502 503 504 /50x.html;
        location = /50x.html {
        }
    }
  子配置文件 /etc/nginx/conf.d/default.conf   
  server {             一个网站，域名，端口，站点目录
    listen       80;     监听端口
    server_name  localhost;   网站的域名，多个空格分割
    location / {             用于匹配用户请求的uri，
        root   /usr/share/nginx/html;  用于指定站点目录
        index  index.html index.htm;   指定首页文件，只输入域名的时候默认展示的页面
    }
    error_page   500 502 503 504  /50x.html;   状态码对应的页面
    location = /50x.html {
        root   /usr/share/nginx/html;
    }
}

    debug，info，notice，warn,error,crit,alert,or emerg

	  location / {
		index index.html;
	  }
	}

8，location规则   匹配用户请求中的URI
   URI：域名后面的内容  /gwp/gwp.avi
   URL：网址   https://www.baidu.com/gwp/gwp.avi
	server {
	  listen 80;
	  server_name gailun.com;
	  root /app/code/gailun/;
	  location / {
		index index.html;
	  }
	  location /admin/ {    
		allow 172.16.1.0/24;
		deny all;
	  }
	  location ~* \.(html|js|css)$ {    ~* 支持正则，不区分大小写   ~支持正则，区分大小写 /x/
		expires 1d;    缓存1天
	  }
	  location ~* \.(jpg|png) {
		expires max;
	  }
	}
	匹配优先级 高到低  =  ^~  ~ ~* /x/
